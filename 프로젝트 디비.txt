CREATE TABLE user(
 n_id INT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
 u_id VARCHAR(255) NOT NULL UNIQUE,
 password VARCHAR(255) NOT NULL,
 name VARCHAR(255) NOT NULL
);

CREATE TABLE Schedule(
 s_id INT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
 title VARCHAR(255) NOT NULL,
 content VARCHAR(1000) NOT NULL,
 year VARCHAR(100) NOT NULL,
 month VARCHAR(100) NOT NULL,
 day VARCHAR(100) NOT NULL,
 userid VARCHAR(255),
 CONSTRAINT fk_calendaruser FOREIGN KEY(userid) REFERENCES user(u_id) ON DELETE CASCADE
);

CREATE TABLE Friends(
 u_id INT UNSIGNED NOT NULL,
 f_id INT UNSIGNED NOT NULL,
 CONSTRAINT pk_friends PRIMARY KEY (u_id,f_id)
);

CREATE TABLE Request(
 r_id INT UNSIGNED NOT NULL,
 u_id INT UNSIGNED NOT NULL,
 CONSTRAINT pk_request PRIMARY KEY (r_id,u_id)
);
